    <script>
        const poems = [
            "Bal gibi tatlı bir sevda, her an seninle, her yerde.",
            "Gözlerin bal rengi, içimdeki nehirleri akar.",
            "Sevdamız bal kadar tatlı, her anı bir ömre bedel.",
            "Bal, senin gibi sıcak ve huzurlu, her gün seninle olmak.",
            "Dudaklarında bal tadı, bir ömrü seninle geçireyim."
        ];

        function generatePoem() {
            const randomPoem = poems[Math.floor(Math.random() * poems.length)];
            document.getElementById("poem").innerText = randomPoem;
        }
    </script>
